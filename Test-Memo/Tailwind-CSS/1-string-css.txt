1. white-space: normal; /* 기본값: 공백과 줄 바꿈을 처리 */
white-space: nowrap; /* 줄 바꿈을 하지 않고 한 줄로 표시 */
white-space: pre; /* 공백과 줄 바꿈을 그대로 유지 */
white-space: pre-wrap; /* 공백과 줄 바꿈 유지, 컨테이너의 너비를 초과하면 줄 바꿈 */
white-space: pre-line; /* 공백은 무시하되 줄 바꿈은 유지 */
=> 줄바꿈이 존재한다면 컨테이너 크기에 따라 변경되는 것입니다.
=> 줄바꿈이 존재하고 텍스트의 길이가 크다면 height을 뚫고 밑으로 내려갑니다.


2. word-wrap: normal; /* 기본값: 단어가 긴 경우 줄 바꿈하지 않음 */
word-wrap: break-word; /* 긴 단어가 컨테이너의 너비를 초과할 경우 줄 바꿈 허용 */
